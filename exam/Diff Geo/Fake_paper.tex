    \begin{enumerate}
    \item (15分) 求圆柱螺旋线 $r(t)=(a\cos t, a \sin t , bt)$ 的曲率和挠率。
    \item (15分) 不用Gauss―Bonnet公式，证明双曲三角形的面积为$\pi - \alpha - \beta - \gamma$， 其中，$\alpha, \beta, \gamma$是三角形的三个内角。
    \item (20分) 若曲面$z=f(x ) + g(z)$ 是极小曲面，求其表达式。
    \item (25分) 已知$(u,0,f(u))$ 为欧式空间$E^3$中曲线，其中 $u>0,f(u)>0$,它绕$z$轴旋转一周得到旋转曲面。试求出$f(u)$的一个表达式使得对应旋转曲面的Gauss曲率$K=-1$。
    \item (25分) 设曲面$S:r=r(u,v)$无抛物点 (即$LM-N^2 \not = 0$)，$n(u,v)$是曲面$S$的法向量。曲面$\tilde S:= r(u,v)+\lambda n(u,v)$(常数$\lambda$足够小)成为$\tilde S$的平行曲面。
    \begin{enumerate}
        \item $\tilde S$的Gauss曲率为 \[\tilde K= \frac{K}{1-2\lambda H +\lambda^2 K}\].
        \item 试求$\tilde S$的平均曲率 $\tilde H$。(用$\lambda, H,K$表示)
    \end{enumerate}
    \end{enumerate}
